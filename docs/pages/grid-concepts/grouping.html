Row group type

There are 2 row group types.
row and column
For row group type, full row can have only group value.
For column group type, each column can have values and also use aggregations over column property agFn

Row Group Column

There is the way to provide row grouping column.
It is not defined in columns. 


new Grid({
  rowGroupColumn: {
    title: 'Group',
    width: 200,
    checkboxSelection: true,
    headerCheckboxSelection: true,
    editable: false,
    render(params){
      const {
        item
      } = params;

      if(!item.$isGroupRow){
        return `${item.brand} ${item.model}`
      }
    }
  },

Custom Row Group value

There is the way to customize row grouping value.

  "Australia": "au",
  "Austria": "at",
  "Belgium": "be",
  "Canada": "ca",
  "Denmark": "dk",
  "Finland": "fi",
  "France": "fr",
  "Germany": "de",
  "Hong Kong": "hk",
  "Iceland": "",
  "Ireland": "ie",
  "Japan": "ja",
  "Luxembourg": "lu",
  "Netherlands": "nl",
  "Norway": "no",
  "San Marino": "",
  "Singapore": "sg",
  "South Korea": "kr",
  "Sweden": "se",
  "Switzerland": "ch",
  "Taiwan": "",
  "UK": "uk",
  "USA": "us"
}

new Grid({
  groupValueRender(params){
    const {
      el,
      value
    } = params;

    if(!countries[value]){
      return value;
    }

    const flagCls = 'flag-' + countries[value];
    const container = document.createElement('span');
    container.classList.add('group-country-container');

    const flagEl = document.createElement('span');
    flagEl.classList.add('flag', flagCls);
    flagEl.innerHTML = ' '
    container.appendChild(flagEl);

    const valueEl = document.createElement('span');
    valueEl.classList.add('group-country-text');
    valueEl.innerHTML = ` ${value}`;
    container.appendChild(valueEl);

    el.appendChild(container);
