filter

Object

Enables column filter.

Examples


columns: [{
  index: 'age',
  title: 'Age',
  type: 'number',
  filter: {
    header: true,
    emptyText: 'Enter'
  }
},{
...

Example: filter header tooltip


columns: [{
  index: 'age',
  title: 'Age',
  type: 'number',
  filter: {
    header: true,
    tip: [
      'Contains: 30',
      'Less than: <30',
      'More than: >30',
      'Equal: =30',
      'Not Equal: !=30',
      'Composite: <30,>5'
    ].join("")
  }
},{
...

Example: Default filter header field operator


columns: [{
  index: 'age',
  title: 'Age',
  type: 'number',
  filter: {
    header: true,
	operator: '='    
  }
},{
...

Default


undefined
