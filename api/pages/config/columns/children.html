children

Object[]

Grid provides one level of column group.

Example


const columns = [{
  index: 'price',
  filter: true,
  agFn: 'sum',
  type: 'currency'
},{
  title: 'Main',
  children: [{
    index: 'brand',
    type: 'string'
  }, {
    index: 'model',
    type: 'string'
  },{
    index: 'year',
    filter: true,
    type: 'number'
  }]
},{
  index: 'country',
  type: 'string'
},{
  index: 'color',
  filter: true,
  type: 'string'
},{
  title: 'Tech',
  children: [{
    index: 'fuelType',
    filter: true,
    type: 'string'
  },{
    index: 'transmission',
    type: 'string'
  },{
    index: 'electric',
    type: 'boolean'
  }]
}]

Default


undefined
