Custom Components

Grid provides a simple way to create custom components.
To do this, use the render property in the column definition.

document.addEventListener('DOMContentLoaded', () => {
  const grid = new Grid({
    renderTo: 'container',
    data: data,
    rowStyle(params){
      if(params.item.fuelType === 'Hybrid'){
        return {
          'background-color': 'rgba(111, 178, 112, 0.2)'
        }
      }
    },
    columns: [{
      type: 'order'
    },{
      title: 'Car',
      width: 160,
      checkboxSelection: true,
      headerCheckboxSelection: true,
      editable: true,
      filter: {
        list: true,
        defaultFilter: 'list'
      },
      type: 'string',
      render(params){
        const brand = params.item.brand || 'No Brand';
        const model = params.item.model || '';

        return `${brand} ${model}`;
      },
      getter(params) {
        const brand = params.item.brand || 'No Brand';
        const model = params.item.model || '';

        return `${brand} ${model}`;
      },
      setter(params) {
        const [brand, ...rest] = params.newValue.split(' ');

        params.item.brand = brand;
        params.item.model = rest.join(' ');
      }
    },{
      index: 'price',
      editable: true,
      filter: true,
      type: 'currency'
    },{
      index: 'color',
      editable: true,
      filter: {
        list: true,
        defaultFilter: 'list'
      },
      type: 'string'
    },{
      index: 'country',
      editable: true,
      filter: {
        list: true,
        defaultFilter: 'list'
      },
      type: 'string'
    },{
      index: 'fuelType',
      width: 120,
      filter: {
        list: true,
        defaultFilter: 'list'
      },
      type: 'string',
      render(params) {
        const divValue = document.createElement('div');
        const buttonRandom = document.createElement('button');

        divValue.innerHTML = params.value;
        buttonRandom.innerHTML = 'rand';

        buttonRandom.addEventListener('click', () => {
          const fuelTypes = ['Gasoline', 'Diesel', 'Hybrid'];
          const randomIndex = Math.floor(Math.random() * fuelTypes.length);
          const randomFuelType = fuelTypes[randomIndex];

          grid.setById(params.item.id, {
            fuelType: randomFuelType
          });
        })

        params.cell.classList.add('fuel-cell');
        params.cell.appendChild(divValue);
        params.cell.appendChild(buttonRandom);
      }
    },{
      index: 'year',
      editable: false,
      filter: {
        list: true,
        defaultFilter: 'list'
      },
      type: 'number',
      render: function(params) {
        const buttonMinus = document.createElement('button');
        const buttonPlus = document.createElement('button');
        const divValue = document.createElement('div');

        buttonMinus.addEventListener('click', () => {
          grid.setById(params.item.id, {
            year: params.value - 1
          });
        });

        buttonPlus.addEventListener('click', () => {
          grid.setById(params.item.id, {
            year: params.value + 1
          });
        });

        buttonMinus.innerHTML = '-';
        buttonPlus.innerHTML = '+';
        divValue.innerHTML = params.value;

        params.cell.classList.add('year-cell');

        params.cell.appendChild(buttonMinus);
        params.cell.appendChild(divValue);
        params.cell.appendChild(buttonPlus);
      }
    },{
      index: 'transmission',
      filter: {
        list: true,
        defaultFilter: 'list'
      },
      editable: true,
      type: 'string'
    },{
      menu: false,
      width: 70,
      render(params){
        const button = document.createElement('button');
        button.innerHTML = `Delete`;
        button.addEventListener('click', () => {
          grid.remove(params.item)
        });

        params.cell.classList.add('delete-cell');
        params.cell.appendChild(button);
      }
    }]
  });
});
